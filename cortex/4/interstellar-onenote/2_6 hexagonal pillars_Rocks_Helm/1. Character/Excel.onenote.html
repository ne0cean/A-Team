Sub SaveAsValue()

    Dim wksht As Worksheet

    For Each wksht In ThisWorkbook.Worksheets

        wksht.Cells.Copy

        wksht.Cells.PasteSpecial xlPasteValues

    Next

    Application.CutCopyMode = False

End Sub
