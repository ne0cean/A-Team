mobile_NA <- mobile


# 1) NA확인

summary(mobile_NA)


# 2)

is.na(mobile_NA) # NA를 TRUE로 반환 (vector 표기를 통해 누락값에 대한 요소 보고)

head(is.na(mobile_NA))

anyNA(mobile_NA) # TRUE / FALSE만 반환 (누락값에 대한 여부만 보고)


sum(is.na(mobile_NA)) # NA의 수 확인

sum(anyNA(mobile_NA)) # NA의 유무만 따지기 때문에 NA값이 여러개가 있더라도 1 값만 반환

sum(is.na(mobile_NA$ARPU)) # 각 변수별 NA 수 파악을 위해 사용


## (2-2) NA 처리

na.omit(mobile_NA) # NA가 있는 행(개체)을 삭제

head(na.omit(mobile_NA))


mobile_NA_omit <- na.omit(mobile_NA) ## NA 처리한 데이터를 mobile_NA_omit로 정의

head(mobile_NA_omit)


summary(mobile_NA_omit)


#5.

#-----------------------------------------------------------------------------------------------

# Outlier 처리

#-----------------------------------------------------------------------------------------------

## (1) 그래프 탐색

boxplot(mobile_NA_omit$ARPU)

summary(mobile_NA_omit$ARPU)

quantile(mobile_NA_omit$ARPU)


## (2) 데이터 확인

mobile_box <- boxplot(mobile$ARPU, horizontal = TRUE, col = "lightblue")

str(mobile_box)

mobile_box$out

text(mobile_box$out, rep(1, NROW(mobile_box$out)), labels = mobile_box$out, pos = c(1,1,3,1))

##-- text(데이터, y좌표, option...)

##-- rep: replicate

##-- labels: 표시할 라벨

##-- pos: position (1: 아래, 3: 위)


head(subset(mobile, ARPU > 80000)) # 이상치만 선택

dim(subset(mobile, ARPU > 80000 ))


mobile_box_out <- subset(mobile_NA_omit, ARPU < 80000) # 범위를 반대로 하면 이상치를 제거한 데이터만 추출 가능

head(mobile_box_out)

dim(mobile_box_out) # 추후 iris_box_out 기반으로 데이터 분석 진행
